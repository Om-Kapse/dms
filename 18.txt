18. Write a PL/SQL program using: While loop DECLARE
  v_number NUMBER := 1;  -- initialize counter
BEGIN
  WHILE v_number <= 5 LOOP
    IF MOD(v_number, 2) = 0 THEN
      DBMS_OUTPUT.PUT_LINE(v_number || ' is even');
    ELSE
      DBMS_OUTPUT.PUT_LINE(v_number || ' is odd');
    END IF;
    v_number := v_number + 1;  -- increment counter
  END LOOP;
END;
/
