22. Write a PL/SQL program based on: Pre-defined Exception Handling
DECLARE
  v_num1 NUMBER := 10;
  v_num2 NUMBER := 0;  -- divisor is zero to trigger exception
  v_result NUMBER;
BEGIN
  -- Attempt division
  v_result := v_num1 / v_num2;
  DBMS_OUTPUT.PUT_LINE('Result: ' || v_result);

EXCEPTION
  -- Pre-defined exception for division by zero
  WHEN ZERO_DIVIDE THEN
    DBMS_OUTPUT.PUT_LINE('Error: Division by zero is not allowed.');

  -- Pre-defined exception for no data found (not used here, just for demo)
  WHEN NO_DATA_FOUND THEN
    DBMS_OUTPUT.PUT_LINE('Error: No data found.');
  
  -- Handle any other unexpected exceptions
  WHEN OTHERS THEN
    DBMS_OUTPUT.PUT_LINE('An unexpected error occurred.');
END;
/
